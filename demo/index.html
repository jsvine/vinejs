This is a demonstration of vine.js by @jsvine.
Recent vines tagged "#cats" on Twitter.
var search = new Vine.Search("#cats");
Recent vines tagged "#cats" or "#dogs" on Twitter.
var search = new Vine.Search("#cats OR #dogs");
Recent cats and dogs within 10 miles of downtown Denver.
var search = new Vine.Search({ q: "cat OR dog", geocode: "39.730426,-104.927673,10mi" });
Vines mentioning "biden" before February 14, 2013.
var search = new Vine.Search({ q: "biden", until: "2013-02-14" });
